7 /*! The base class for packet-listeners, i.e., all participants of protocols.
8 The receiving half of the actor model.
9 
10 Each instance of `TransportReceiver` has a static address throughout its
11 lifetime. It also can only work with one specific `Transport` type.
12 
13 After registered to a `Transport` instance, virtual method `receiveMessage` will
14 be called whenever messages arrives receiver's address.
15 
16 The `TransportReceiver` itself does not make any assumption on how to process
17 received message, but a subclass will probably want to call `Transport` methods
18 to send message and schedule timeout. So it is conventional to ask for a
19 reference of `Transport` instance when constructing subclass. Because the
20 `Transport` base class follows CRTP, `Transport &` is basically equals to
21 `oscar::Transport<Transport> &` in every situation.
22 */
23 template <typename Transport> class TransportReceiver
24 {
25 public:
26  //! Static receiving address of the receiver. Must be unique across the
27  //! whole system.
28  const typename Transport::Address address;
29 
30  //! Construct receiver base class with address.
31  //!
32  TransportReceiver(typename Transport::Address address) : address(address) {}
33  virtual ~TransportReceiver() {}
34 
35  //! Handle received raw message. The underlying memory that backs `span`
36  //! will go out of lifetime after this method returns, so receiver need to
37  //! own everything necessary for later processing.
38  virtual void
39  receiveMessage(const typename Transport::Address &remote, Span span) = 0;
40 
41  //! If receiver is registered to multicast as well, multicast message will
42  //! be sent to here. This gives receiver a chance to determine multicast
43  //! messages.
44  // the codebase assume at most one multicast address present
45  virtual void receiveMulticastMessage(
46  const typename Transport::Address &remote, Span span)
47  {
48  receiveMessage(remote, span);
49  }
50 };
