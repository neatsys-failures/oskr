99 template <typename Transport, typename ReplicaMessage>
100 void BasicClient<Transport, ReplicaMessage>::invoke(
101  Data op, typename BasicClient::InvokeCallback callback)
102 {
103  if (pending) {
104  panic("Invoke pending client");
105  }
106 
107  request_number += 1;
108  pending = PendingRequest{request_number, op, {}, callback};
109  sendRequest(false);

112 template <typename Transport, typename ReplicaMessage>
113 void BasicClient<Transport, ReplicaMessage>::sendRequest(bool resend)
114 {
115  RequestMessage request;
116  request.client_id = this->client_id;
117  request.request_number = pending->request_number;
118  request.op = pending->op;
119  auto write = [this, request](auto &buffer) {
120  return serializeRequestMessage(buffer, ReplicaMessage(request));
121  };
122  auto send_to_primary = [this, write] {
123  transport.sendMessageToReplica(
124  *this, transport.config.primary_id(view_number), write);
125  };
126  auto send_to_all = [this, write] {
127  transport.sendMessageToAll(*this, write);
128  };
129 
130  switch (config.strategy) {
131  case Config::Strategy::ALL:
132  send_to_all();
133  break;
134  case Config::Strategy::PRIMARY_FIRST:
135  if (resend) {
136  send_to_all();
137  } else {
138  send_to_primary();
139  }
140  break;
141  default:
142  panic("Unreachable");
143  }
144 
145  transport.scheduleTimeout(
146  config.resend_interval, [this, current_number = request_number] {
147  if (!pending || request_number != current_number) {
148  return;
149  }
150  warn("Resend: request number = {}", request_number);
151  sendRequest(true);
152  });
153 }

155 template <typename Transport, typename ReplicaMessage>
156 void BasicClient<Transport, ReplicaMessage>::handleReply(
157  const ReplyMessage &reply)
158 {
159  if (!pending) {
160  return;
161  }
162  if (pending->request_number != reply.request_number) {
163  return;
164  }
165 
166  if (reply.view_number > view_number) {
167  view_number = reply.view_number;
168  }
169 
170  if (config.n_matched > 1) {
171  pending->result_table[reply.result].insert(reply.replica_id);
172  if (pending->result_table.at(reply.result).size() < config.n_matched) {
173  return;
174  }
175  }
176 
177  auto callback = pending->callback;
178  pending.reset();
179  callback(reply.result);
180 }
