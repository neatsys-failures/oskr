92 template <typename Sender>
93 void Simulated::sendMessage(
94  const Sender &sender, const Address &dest, Write write)
95 {
96  if (!receiver_table.count(dest)) {
97  panic("Send to unknown destination: sender = {}", sender.address);
98  }
99 
100  microseconds delay = 0us;
101  for (auto pair : filter_table) {
102  if (!pair.second(sender.address, dest, delay)) {
103  info("Message dropped: filter id = {}", pair.first);
104  return;
105  }
106  }
107  if (delay != 0us) {
108  info("Message delayed: {}us", delay.count());
109  }
110 
111  Data message(BUFFER_SIZE);
112  message.resize(write(TxSpan<BUFFER_SIZE>(message.data(), BUFFER_SIZE)));
113  destiny_queue.insert({{now_us + delay.count(), [&, dest, message] {
114  channel_id = -2;
115  // TODO multicast
116  receiver_table.at(dest)(sender.address, message);
117  }}});
118 }

120 void Simulated::run(microseconds time_limit)
121 {
122  while (true) {
123  if (microseconds(now_us) >= time_limit) {
124  panic(
125  "Hard time limit reached: {}ms",
126  double(time_limit.count()) / 1000);
127  }
128 
129  auto timeout_iter = destiny_queue.begin();
130  if (timeout_iter == destiny_queue.end()) {
131  return;
132  }
133  now_us = timeout_iter->first;
134  timeout_iter->second();
135  destiny_queue.erase(timeout_iter);
136  }
137 }
