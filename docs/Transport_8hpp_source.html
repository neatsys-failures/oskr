29 /*! General supporting runtime for `TransportReceiver`, the sending half of the
30 actor model (and more).
31 
32 The base class takes curiously recurring template pattern. The subclass
33 `Transport` is passed as template argument to `Transport`, as well as `Config`,
34 `TransportReceiver`, etc.
35 
36 Besides of all interfaces defined here, a `Transport` implementation probably
37 wants to provide a `run` method to start main loop.
38 */
39 template <typename Self> class Transport
40 {
41 protected:
42  //! Construct base class with global configuration.
43  //!
44  Transport(const Config<Self> &config) : config(config) {}
45 
46 public:
47  virtual ~Transport() {}
48 
49  //! Value-semantic address type. Possible choices include plain string,
50  //! `socketaddr_in`, or more customized ones.
51  using Address = typename TransportMeta<Self>::Address;
52 
53  //! Dynamically allocate free address, mainly for construct `Client`.
54  //!
55  virtual Address allocateAddress() = 0;
56 
57  //! Configuration for `Transport` instance, along with all registered
58  //! receivers.
59  const Config<Self> &config;
60 
61  //! Register a receiver. The caller ensures that receiver's reference will
62  //! be valid as long as message get received on receiver's address, which
63  //! probably happens through the rest of `Transport`'s lifetime.
64  virtual void registerReceiver(TransportReceiver<Self> &receiver) = 0;
65  //! Register a receiver that listens to multicast messages. Transport
66  //! implementation may require receiver to call `registerReceiver` first.
67  virtual void
68  registerMulticastReceiver(TransportReceiver<Self> &receiver) = 0;
69 
70  //! General task closure. Receiver can keep things inside it alive between
71  //! working steps.
72  using Callback = std::function<void()>;
73 
74  //! Schedule a one-time timeout. `callback` will be called after `delay`,
75  //! cannot be canceled.
76  //!
77  //! Unlike receiving message, timeout is non-direction, i.e., does not have
78  //! a `receiver`. The behavior depends on `callback`'s capturing.
79  //!
80  //! Timeout's `callback` is always executed sequentially.
81  virtual void
82  scheduleTimeout(std::chrono::microseconds delay, Callback callback) = 0;
83 
84  //! Schedule a sequential task. Sequential task is promised to be executed
85  //! one by one without concurrency, so different sequential tasks can share
86  //! states without locking. However, it is not guaranteed that all
87  //! sequential tasks will be executed by the same thread.
88  //!
89  //! A default implementation is provided mainly for implying that timeout is
90  //! sequential task by default, so there is no need for calling
91  //! `scheduleSequential` inside timeout callback.
92  virtual void scheduleSequential(Callback callback)
93  {
94  scheduleTimeout(std::chrono::microseconds(0), callback);
95  }
96 
97  //! Schedule a concurrent task. Multiple concurrent tasks may be executed
98  //! with temporal overlapping, so they would better to be stateless.
99  virtual void scheduleConcurrent(Callback callback) = 0;
100 
101  //! Every concurrent task get assigned to a concurrent id, which can be
102  //! queried by calling `getConcurrentId` during execution. Concurrent id is
103  //! unsigned and bounded.
104  //!
105  //! Transport implementation promise that all concurrent tasks that get
106  //! assigned to the same concurrent id will be sequentially executed. So it
107  //! is safe to reuse pre-allocated memory "slot" among those tasks.
108  //!
109  //! Transport may set concurrent id to `-1` for sequential tasks, but it is
110  //! encouraged to determine sequential tasks by themselves.
111  virtual int getConcurrentId() const = 0;
112 
113  //! The size of buffer of tx message. May be helpful in assertion.
114  //!
115  static constexpr std::size_t BUFFER_SIZE = TransportMeta<Self>::BUFFER_SIZE;
116 
117  //! Argument for `sendMessage*` methods. The lambda should in-place
118  //! serialize message into `buffer`, and return message length. The returned
119  //! value must not exceed `BUFFER_SIZE`.
120  using Write = std::function<std::size_t(Buffer<BUFFER_SIZE> &buffer)>;
121 
122  //! Send a message. The first argument is probably `*this`.
123  //!
124  //! Transport implementation promise not to access `write` after this method
125  //! return, so the closure can capture references with no worry.
126  virtual void sendMessage(
127  const TransportReceiver<Self> &sender, const Address &dest,
128  Write write) = 0;
129 
130  void sendMessageToReplica(
131  const TransportReceiver<Self> &sender, ReplicaId replica_id,
132  Write write)
133  {
134  sendMessage(sender, config.replica_address_list[replica_id], write);
135  }
136 
137  //! The default implementation is mainly for illustration purpose. Transport
138  //! should provide a more efficient version that only serialize once.
139  void sendMessageToAll(const TransportReceiver<Self> &sender, Write write)
140  {
141  for (auto address : config.replica_address_list) {
142  if (address != sender.address) {
143  sendMessage(sender, address, write);
144  }
145  }
146  }
147 
148  virtual void
149  sendMessageToMulticast(const TransportReceiver<Self> &sender, Write write)
150  {
151  sendMessage(sender, config.multicast_address, write);
152  }
153 };
